pipeline {
    agent any

    stages {

        stage('Clone Code') {
            steps {
                git 'https://github.com/subasinik-blip/aws-devops-project.git'
            }
        }

        stage('Install Dependencies') {
            steps {
                dir('app') {
                    sh 'npm install'
                }
            }
        }

        stage('Deploy Application') {
            steps {
                dir('app') {
                    sh 'nohup node server.js > output.log 2>&1 &'
                }
            }
        }

        stage('Deployment Complete') {
            steps {
                echo "Application Deployed Successfully ðŸš€"
            }
        }
    }
}
